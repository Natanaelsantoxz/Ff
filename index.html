/* ========================================================= MINHAS FINANÇAS — lógica do app (100% local, sem servidor) Dados salvos no localStorage do navegador. ========================================================= */ const STORAGE_KEY = "financas_transacoes"; const STORAGE_GOALS = "financas_metas"; const ICONS = ["💰","🏦","💸","🧾","🍔","🏠","🚗","💊","🎓","🎮","✈️","🛒", "📱","💡","🎬","👕","🐶","🏋️","☕","🎁","📈","🧴","🚌","🎯"]; let transacoes = JSON.parse(localStorage.getItem(STORAGE_KEY) || "[]"); } function formatarMoeda(v){ return v.toLocaleString("pt-BR", { style:"currency", currency:"BRL" }); } function formatarData(iso){ if(!iso) return "-"; const [ano,mes,dia] = iso.split("-"); return `${dia}/${mes}/${ano}`; document.querySelectorAll(".view").forEach(v=>v.classList.remove("active")); btn.classList.add("active"); document.getElementById("view-"+btn.dataset.view).classList.add("active"); }); }); /* ---------- SELETOR DE ÍCONES ---------- */ document.querySelectorAll(".icon-picker").forEach(picker=>{ const optionsBox = document.createElement("div"); optionsBox.className = "icon-options"; ICONS.forEach(ic=>{ const b = document.createElement("button"); b.type = "button"; b.textContent = ic; b.addEventListener("click", ()=>{ const current = picker.querySelector(".icon-current"); salvar(); form.reset(); dateInput.valueAsDate = new Date(); const iconTarget = form.querySelector('input[type="hidden"]'); const defaultIcon = form.dataset.type === "renda_fixa" ? Continuar?")){ transacoes = []; metas = []; salvar(); render(); } }); /* ---------- RENDERIZAÇÃO DAS TABELAS ---------- */ function linhaTransacao(t, comAcao){ const sinal = t.tipo === "gasto" ? "-" : "+"; const classeValor = t.tipo === "gasto" ? "valor-neg" : "valor-pos"; const tagLabel = { renda_fixa:"Renda fixa", renda_extra:"Renda extra", gasto:"Gasto" }[t.tipo]; return ` ${t.icone} ${t.descricao} ${t.categoria} ${comAcao ? "" : `${tagLabel}`} ${formatarData(t.data)} ${sinal} ${formatarMoeda(Math.abs(t.valor))} ${comAcao ? `` : ""} `; } function renderTabelaPorTipo(tipo){ const tbody = document.querySelector(`#tabela-${tipo} tbody`); const lista = transacoes.filter(t=>t.tipo === tipo).sort((a,b)=> new Date(b.data)-new Date(a.data)); if(lista.length === 0){ tbody.innerHTML = `Nenhum lançamento ainda. Adicione o primeiro acima ✨`; return; } tbody.innerHTML = lista.map(t=>linhaTransacao(t, true)).join(""); } function renderRecentes(){ const tbody = document.querySelector("#tabelaRecentes tbody"); const lista = [...transacoes].sort((a,b)=> new Date(b.data)-new Date(a.data)).slice(0,8); if(lista.length === 0){ tbody.innerHTML = `Nenhuma movimentação ainda.`; return; } tbody.innerHTML = lista.map(t=>linhaTransacao(t, false)).join(""); } /* ---------- RENDERIZAÇÃO DOS CARDS ---------- */ function renderCards(){ const totalFixa = transacoes.filter(t=>t.tipo==="renda_fixa").reduce((s,t)=>s+t.valor,0); const totalExtra = transacoes.filter(t=>t.tipo==="renda_extra").reduce((s,t)=>s+t.valor,0); const totalGastos = transacoes.filter(t=>t.tipo==="gasto").reduce((s,t)=>s+t.valor,0); const saldo = totalFixa + totalExtra - totalGastos; document.getElementById("cardRendaFixa").textContent = formatarMoeda(totalFixa); document.getElementById("cardRendaExtra").textContent = formatarMoeda(totalExtra); document.getElementById("cardGastos").textContent = formatarMoeda(totalGastos); document.getElementById("cardSaldo").textContent = formatarMoeda(saldo); document.getElementById("cardSaldo").style.color = saldo >= 0 ? "" : "var(--red)"; } /* ---------- RENDERIZAÇÃO DAS METAS ---------- */ function renderMetas(){ const grid = document.getElementById("goalsGrid"); if(metas.length === 0){ grid.innerHTML = `

Você ainda não criou nenhuma meta. Que tal começar agora? 🎯

`; return; } grid.innerHTML = metas.map(m=>{ const pct = m.alvo > 0 ? Math.min(100, (m.atual/m.alvo)*100) : 0; return `
${m.icone}
${m.nome}
${formatarMoeda(m.atual)} de ${formatarMoeda(m.alvo)} ${pct.toFixed(0)}%
`; }).join(""); } /* ---------- GRÁFICOS ---------- */ let chartBarras, chartRosca; function ultimosMeses(n){ const arr = []; const hoje = new Date(); for(let i=n-1; i>=0; i--){ const d = new Date(hoje.getFullYear(), hoje.getMonth()-i, 1); } function renderGraficoRosca(){ const gastos = transacoes.filter(t=>t.tipo==="gasto"); const porCategoria = {}; gastos.forEach(t=>{ porCategoria[t.categoria] = (porCategoria[t.categoria]||0) + t.valor; }); const cores = ["#1e3a8a","#2563eb","#3b82f6","#60a5fa","#93c5fd","#bfdbfe","#dbeafe"]; const ctx = document.getElementById("chartRosca"); if(chartRosca) chartRosca.destroy(); const labels = Object.keys(porCategoria); if(labels.length === 0){ chartRosca = new Chart(ctx, { type:"doughnut", data:{ labels:["Sem gastos ainda"], datasets:[{ data:[1], backgroundColor:["#e8f0fe"] }] }, options:{ plugins:{ legend:{ display:false }, tooltip:{ enabled:false } } } }); return; } chartRosca = new Chart(ctx, { type: "doughnut", data: { labels, datasets: [{ data: Object.values(porCategoria), backgroundColor: cores, borderWidth:2, borderColor:"#fff" }] }, options: { responsive:true, plugins:{ legend:{ position:"bottom", labels:{ usePointStyle:true, boxWidth:8 } } } } }); } /* ---------- RENDER GERAL ---------- */ function render(){ renderCards(); renderRecentes(); renderTabelaPorTipo("renda_fixa"); renderTabelaPorTipo("renda_extra"); renderTabelaPorTipo("gasto"); renderMetas(); renderGraficoBarras(); renderGraficoRosca(); } render();
